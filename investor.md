## Ce qu'on construit

Une plateforme de Due Diligence IA **pour Business Angels** qui :

1. **Fait le travail d'un analyste** - En 1h, pas en 2 jours. Tu es seul mais tu as la puissance d'une equipe.

2. **Donne acces aux donnees pro** - 50K+ deals, benchmarks, comparables. Les memes donnees que les VCs, sans payer 20K€/an.

3. **Detecte les red flags** - Le fondateur a une startup ratee qu'il cache ? On le trouve. CV gonfle ? On le detecte.

4. **Prepare la negociation** - "La valo est 30% au-dessus du marche, voici les arguments pour negocier."

5. **Donne confiance** - Plus de "je sais pas trop, ca a l'air bien". Des decisions basees sur des donnees.

```typescript
          weight: 0.40;

        };

        technology_readiness: {
          weight: 0.30;
          factors: ['enabling_tech_maturity', 'infrastructure_availability'];
        };

        regulatory_environment: {
          weight: 0.30;
          factors: ['regulatory_clarity', 'upcoming_changes', 'compliance_burden'];
        };
      };
    };

    // 5. FINANCIAL SCORE
    financials: {
      weight: 0.20;

      components: {
        valuation_reasonableness: {
          weight: 0.40;
          method: 'compare_to_similar_deals';
          output: {
            percentile: number;      // "85e percentile = cher"
            vs_median: string;       // "+67% vs median"
            verdict: 'cheap' | 'fair' | 'expensive' | 'very_expensive';
          };
        };

        capital_efficiency: {
          weight: 0.25;
          factors: ['burn_multiple', 'arr_per_employee', 'runway'];
        };

        round_structure: {
          weight: 0.20;
          factors: ['dilution', 'liquidation_prefs', 'governance'];
        };

        path_to_next_round: {
          weight: 0.15;
          factors: ['milestones_achievable', 'runway_vs_milestones', 'follow_on_appetite'];
        };
      };
    };
  };

  // ═══════════════════════════════════════════════════════════════════════════
  // SCORE GLOBAL & POSITIONING
  // ═══════════════════════════════════════════════════════════════════════════

  globalScore: {

    calculate: (dimensionScores: DimensionScores) => {
      // Weighted average
      const weighted = Object.entries(dimensionScores).reduce((sum, [dim, score]) => {
        return sum + (score.score * dimensions[dim].weight);
      }, 0);

      return {
        score: Math.round(weighted),
        grade: scoreToGrade(weighted),

        // Breakdown
        dimensions: dimensionScores,

        // Strengths & concerns (top 3 each)
        top_strengths: extractTopStrengths(dimensionScores, 3),
        top_concerns: extractTopConcerns(dimensionScores, 3),

        // The money insight
        one_liner: generateOneLiner(dimensionScores)
        // Ex: "Strong team, fair valuation, but market timing is concerning"
      };
    };
  };

  // ═══════════════════════════════════════════════════════════════════════════
  // POSITIONNEMENT COMPARATIF
  // ═══════════════════════════════════════════════════════════════════════════

  comparativePosition: {

    calculate: async (deal: Deal, globalScore: number) => {
      // Trouver les deals comparables
      const comparables = await findComparableDeals({
        sector: deal.sector,
        stage: deal.stage,
        geography: deal.geography,
        time_window: '24_months'
      });

      // Calculer le percentile
      const percentile = calculatePercentile(globalScore, comparables.map(d => d.score));

      // Trouver les deals les plus similaires
      const mostSimilar = comparables
        .sort((a, b) => Math.abs(a.score - globalScore) - Math.abs(b.score - globalScore))
        .slice(0, 3);

      return {
        percentile: percentile,
        position_text: `Top ${100 - percentile}% des ${comparables.length} deals ${deal.stage} ${deal.sector}`,

        // Comparables proches
        similar_deals: mostSimilar.map(d => ({
          name: d.anonymized_name,  // "SaaS B2B #127"
          score: d.score,
          outcome: d.outcome,       // "Series A x2.5" ou "Echec"
          key_similarity: d.similarity_reason
        })),

        // Insight
        insight: generateComparativeInsight(deal, comparables, percentile)
        // Ex: "Les 5 deals avec un score similaire (75-80) ont leve Series A dans 80% des cas"
      };
    };
  };
}
```

## FEATURE 2 : Red Flags Automatiques

> **"Detection proactive. Zero faux positifs. Confiance totale."**

### Philosophie

```
Le probleme des red flags:
- Trop de faux positifs → L'utilisateur arrete de les regarder
- Pas assez → On rate des deals toxiques

Notre approche:
- Chaque red flag a un NIVEAU DE CONFIANCE
- On ne montre que les red flags VERIFIES ou FORTEMENT SUSPECTES
- Chaque flag vient avec la SOURCE et la METHODE de detection
```

## FEATURE 3 : ROI Simulator

```typescript
interface ROISimulator {

  // ═══════════════════════════════════════════════════════════════════════════
  // EXIT SCENARIOS
  // ═══════════════════════════════════════════════════════════════════════════

  exitScenarios: {

    generate: async (deal: Deal, investment: Investment) => {

      // Recuperer les exits comparables
      const comparableExits = await getComparableExits({

        growth_then_acquisition: {

          typical_acquirers: ['Big Tech', 'Large strategics', 'PE'],
          comparable_exits: comparableExits.filter(e => e.type === 'acquisition_large'),
        },

        // Scenario 3: IPO track
        ipo: {
          probability_bucket: 'unlikely_but_possible',
          timeline: '7-10 ans',
          exit_value_range: { low: 500_000_000, mid: 1_500_000_000, high: 5_000_000_000 },
          multiple_on_investment: { low: 20, mid: 50, high: 150 },
          dilution_to_exit: '80-90%',
          requirements: ['$100M+ ARR', 'Path to profitability', 'Market leadership'],
          comparable_exits: comparableExits.filter(e => e.type === 'ipo'),
        },

        // Scenario 4: Echec
        failure: {
          probability_bucket: 'possible',
          timeline: '2-4 ans',
          exit_value_range: { low: 0, mid: 0, high: 500_000 },
          multiple_on_investment: { low: 0, mid: 0, high: 0.1 },
          common_causes: ['Market timing', 'Competition', 'Execution', 'Funding gap'],
          comparable_failures: comparableExits.filter(e => e.type === 'failure'),
        },
      };
    };
  };

  // ═══════════════════════════════════════════════════════════════════════════
  // DILUTION PATH
  // ═══════════════════════════════════════════════════════════════════════════

  dilutionPath: {

    project: (deal: Deal, investment: Investment) => {

      // Hypotheses basees sur les rounds typiques
      const rounds = [
        { name: 'Seed (current)', amount: deal.raise_amount, dilution: deal.dilution },
        { name: 'Series A', amount: 8_000_000, dilution: 0.20, typical_timing: '+18 mois' },
        { name: 'Series B', amount: 25_000_000, dilution: 0.18, typical_timing: '+36 mois' },
        { name: 'Series C', amount: 60_000_000, dilution: 0.15, typical_timing: '+54 mois' },
      ];

      // Calculer l'ownership a chaque etape
      let yourOwnership = investment.amount / deal.post_money;

      const path = rounds.map(round => {
        const result = {
          round: round.name,
          your_ownership: yourOwnership,
          your_stake_value: yourOwnership * round.post_money,
        };

        yourOwnership = yourOwnership * (1 - round.dilution);
        return result;
      });

      return {
        path: path,
        final_ownership_range: { low: yourOwnership * 0.8, high: yourOwnership * 1.2 },
        insight: `Votre ${investment.amount}€ represente ${(investment.amount / deal.post_money * 100).toFixed(1)}% aujourd'hui,
                  probablement ${(yourOwnership * 100).toFixed(1)}% a l'exit apres dilution.`
      };
    };
  };

  // ═══════════════════════════════════════════════════════════════════════════
  // COMPARABLE EXITS (Real Data)
  // ═══════════════════════════════════════════════════════════════════════════

  comparableExits: {

    display: (exits: Exit[]) => {
      return exits.map(exit => ({
        company: exit.name,
        sector: exit.sector,
        entry_stage: exit.entry_stage,
        entry_valuation: exit.entry_valuation,
        exit_type: exit.exit_type,
        exit_value: exit.exit_value,
        time_to_exit: exit.time_to_exit,
        multiple: exit.exit_value / exit.entry_valuation,
        acquirer: exit.acquirer,

        // What made it work/fail
        key_factors: exit.key_factors,

        // Similarity to current deal
        similarity_score: exit.similarity_score,
        similarity_reasons: exit.similarity_reasons,
      }));
    };
  };

  // ═══════════════════════════════════════════════════════════════════════════
  // MONTE CARLO DISTRIBUTION
  // ═══════════════════════════════════════════════════════════════════════════

  monteCarlo: {

    simulate: (deal: Deal, investment: Investment, simulations: number = 10000) => {
      // Distribution basee sur les outcomes historiques de deals similaires

      const results = [];
      const historicalOutcomes = getHistoricalOutcomeDistribution(deal.sector, deal.stage);

      for (let i = 0; i < simulations; i++) {
        const outcome = sampleFromDistribution(historicalOutcomes);
        const dilution = sampleDilutionPath(deal.stage, outcome.exit_stage);
        const exitMultiple = sampleExitMultiple(outcome.exit_type, deal.sector);

        const finalValue = investment.amount * exitMultiple * (1 - dilution);
        results.push(finalValue);
      }

      return {
        distribution: computeDistribution(results),
        percentiles: {
          p10: percentile(results, 10),
          p25: percentile(results, 25),
          p50: percentile(results, 50),
          p75: percentile(results, 75),
          p90: percentile(results, 90),
        },
        expected_value: mean(results),
        probability_of_loss: results.filter(r => r < investment.amount).length / simulations,
        probability_of_2x: results.filter(r => r > investment.amount * 2).length / simulations,
        probability_of_10x: results.filter(r => r > investment.amount * 10).length / simulations,
      };
    };
  };
}
```

```typescript
interface ChallengePartner {

  // ═══════════════════════════════════════════════════════════════════════════
  // ASSUMPTION CHECKER
  // ═══════════════════════════════════════════════════════════════════════════

  assumptionChecker: {

    identify: async (deal: Deal) => {

      // Extraire les claims du deck
      const claims = extractClaimsFromDeck(deal.documents);

      // Identifier les assumptions implicites
      const assumptions = [];

      for (const claim of claims) {
        const implicitAssumptions = identifyImplicitAssumptions(claim);

        for (const assumption of implicitAssumptions) {
          assumptions.push({
            claim: claim.text,
            assumption: assumption.text,

            // Est-ce que c'est vrai?
            validation: {
              status: 'unverified' | 'partially_verified' | 'verified' | 'contradicted',
              evidence: string[],
              confidence: number,
            },

            // Qu'est-ce qui se passe si c'est faux?
            risk_if_wrong: {
              impact: 'critical' | 'high' | 'medium' | 'low',
              scenario: string,
            },

            // Comment verifier?
            how_to_verify: string[],
          });
        }
      }

      return assumptions;
    };
  };

  // ═══════════════════════════════════════════════════════════════════════════
  // BLIND SPOT FINDER
  // ═══════════════════════════════════════════════════════════════════════════

  blindSpotFinder: {

    find: async (deal: Deal, analysis: Analysis) => {

      const blindSpots = [];

      // 1. Ce qui n'est PAS dans le deck mais devrait y etre
      const missingTopics = identifyMissingTopics(deal.documents, deal.sector);
      for (const topic of missingTopics) {
        blindSpots.push({
          type: 'missing_from_deck',
          topic: topic.name,
          why_important: topic.importance,
          question: `Le deck ne mentionne pas ${topic.name}. Pourquoi?`,
        });
      }

      // 2. Questions que personne ne pose mais qu'on devrait
      const uncommonQuestions = await getUncommonButImportantQuestions(deal.sector);
      for (const q of uncommonQuestions) {
        blindSpots.push({
          type: 'rarely_asked',
          question: q.question,
          why_important: q.why,
          what_we_learned: q.learning_from_past_deals,
        });
      }

      // 3. Angles morts basés sur les échecs similaires
      const similarFailures = await getSimilarFailures(deal);
      for (const failure of similarFailures) {
        if (failure.warning_signs_visible_in_hindsight) {
          blindSpots.push({
            type: 'learned_from_failure',
            company: failure.name,
            what_happened: failure.failure_reason,
            warning_sign: failure.warning_signs_visible_in_hindsight,
            question: `[${failure.name}] a echoue parce que ${failure.failure_reason}.
                       Comment ce deal evite-t-il le meme piege?`,
          });
        }
      }

      // 4. Ce que le fondateur ne dit pas
      const unsaidThings = identifyWhatFounderIsNotSaying(deal);
      for (const thing of unsaidThings) {
        blindSpots.push({
          type: 'unsaid',
          observation: thing.observation,
          question: thing.question,
          why_suspicious: thing.reasoning,
        });
      }

      return blindSpots;
    };
  };

  // ═══════════════════════════════════════════════════════════════════════════
  // SCENARIO EXPLORER
  // ═══════════════════════════════════════════════════════════════════════════

  scenarioExplorer: {

    explore: async (deal: Deal) => {

      const scenarios = [];

      // Scenario 1: Market shifts
      scenarios.push({
        category: 'market',
        scenarios: [
          {
            name: 'Market decline',
            trigger: 'Le marche se contracte de 30% en 18 mois',
            probability: 'possible',
            impact_on_deal: await analyzeImpact(deal, 'market_decline_30'),
            questions: ['Quel est le plan si le marche se contracte?', 'Quels sont les couts fixes?'],
          },
          {
            name: 'Market acceleration',
            trigger: 'Le marche accelere, nouveaux entrants',
            probability: 'possible',
            impact_on_deal: await analyzeImpact(deal, 'market_acceleration'),
            questions: ['Comment scaler rapidement si le marche explose?', 'Risque de se faire depasser?'],
          },
        ],
      });

            probability: deal.context.competition.big_tech_risk,
            impact_on_deal: 'Severe - differentiation critique',

          },
          {
            name: 'Leader acquires competitor',

            impact_on_deal: await analyzeImpact(deal, 'consolidation'),
            questions: ['Position dans un marche consolide?', 'Cible d\'acquisition?'],
          },
        ],
      });

      // Scenario 3: Team
      scenarios.push({
        category: 'team',
        scenarios: [
          {
            name: 'Founder departure',
            trigger: 'Le CEO/CTO quitte l\'entreprise',
```

```typescript
// ═══════════════════════════════════════════════════════════════════════════
// MODEL ROUTER - Model-Agnostic, Learning-Based
// ═══════════════════════════════════════════════════════════════════════════

interface ModelRouter {

  // ─────────────────────────────────────────────────────────────────────────
  // AVAILABLE MODELS (via OpenRouter)
  // ─────────────────────────────────────────────────────────────────────────

  models: {
    // The router doesn't care what models exist
    // It discovers them from OpenRouter and learns their strengths

    registry: Array<{
      id: string;                    // 'anthropic/claude-3-opus'
      provider: string;              // 'anthropic'
      name: string;                  // 'Claude 3 Opus'

      // Capabilities (from OpenRouter)
      capabilities: {
        context_window: number;
        supports_vision: boolean;
        supports_function_calling: boolean;
        supports_streaming: boolean;
      };

      // Pricing (from OpenRouter)
      pricing: {
        input_per_1k: number;
        output_per_1k: number;
      };

      // Learned performance (our data)
      learned_performance: {
        // Per task type
        by_task: Record<TaskType, {
          sample_size: number;
          avg_quality_score: number;    // 0-100, from human feedback
          avg_latency_ms: number;
          success_rate: number;
          cost_efficiency: number;       // quality / cost
        }>;

        // Overall
        overall_quality: number;
        overall_reliability: number;
        last_evaluated: Date;
      };

      // Status
      status: 'active' | 'degraded' | 'down';
      last_health_check: Date;
    }>;

    // Auto-discovery
    discovery: {
      source: 'openrouter_api';
      refresh_interval: 24 * 60 * 60;  // Daily
      auto_add_new_models: true;
      min_evaluation_before_production: 50;  // samples
    };
  };

  // ─────────────────────────────────────────────────────────────────────────
  // ROUTING STRATEGIES
  // ─────────────────────────────────────────────────────────────────────────

  strategies: {

    // Strategy 1: Best quality (regardless of cost)
    quality_first: {
      select: (task: TaskType, models: Model[]) => {
        return models
          .filter(m => m.status === 'active')
          .filter(m => m.learned_performance.by_task[task]?.sample_size > 20)
          .sort((a, b) =>
            b.learned_performance.by_task[task].avg_quality_score -
            a.learned_performance.by_task[task].avg_quality_score
          )[0];
      };
    };

    // Strategy 2: Best cost efficiency
    cost_optimized: {
      select: (task: TaskType, models: Model[]) => {
        return models
          .filter(m => m.status === 'active')
          .filter(m => m.learned_performance.by_task[task]?.avg_quality_score > 70)
          .sort((a, b) =>
            b.learned_performance.by_task[task].cost_efficiency -
            a.learned_performance.by_task[task].cost_efficiency
          )[0];
      };
    };

    // Strategy 3: Fastest (for interactive features)
    speed_first: {
      select: (task: TaskType, models: Model[]) => {
        return models
          .filter(m => m.status === 'active')
          .filter(m => m.learned_performance.by_task[task]?.avg_quality_score > 60)
          .sort((a, b) =>
            a.learned_performance.by_task[task].avg_latency_ms -
            b.learned_performance.by_task[task].avg_latency_ms
          )[0];
      };
    };

    // Strategy 4: Balanced (default)
    balanced: {
      select: (task: TaskType, models: Model[]) => {
        // Weighted score combining quality, cost, speed
        const score = (m: Model) => {
          const perf = m.learned_performance.by_task[task];
          if (!perf || perf.sample_size < 10) return 0;

          return (
            perf.avg_quality_score * 0.5 +
            perf.cost_efficiency * 0.3 +
            (1000 / perf.avg_latency_ms) * 0.2
          );
        };

        return models
          .filter(m => m.status === 'active')
          .sort((a, b) => score(b) - score(a))[0];
      };
    };

    // Strategy 5: Exploration (try new/under-sampled models)
    exploration: {
      probability: 0.1;  // 10% of requests
      select: (task: TaskType, models: Model[]) => {
        const underSampled = models.filter(m =>
          m.status === 'active' &&
          (m.learned_performance.by_task[task]?.sample_size || 0) < 50
        );

        if (underSampled.length > 0) {
          return underSampled[Math.floor(Math.random() * underSampled.length)];
        }
        return null;  // Fall back to balanced
      };
    };
  };

  // ─────────────────────────────────────────────────────────────────────────
  // TASK TYPES
  // ─────────────────────────────────────────────────────────────────────────

  taskTypes: {
    // Document processing
    'ocr_extraction': { requires_vision: true };
    'claim_extraction': { requires_long_context: true };
    'document_summary': {};

    // Analysis tasks
    'financial_analysis': { complexity: 'high' };
    'market_analysis': { complexity: 'high' };
    'team_analysis': { complexity: 'medium' };
    'competitive_analysis': { complexity: 'high' };
    'legal_analysis': { complexity: 'high' };
    'technical_analysis': { complexity: 'high' };

    // Synthesis tasks
    'contradiction_detection': { requires_reasoning: true };
    'scenario_modeling': { requires_reasoning: true };
    'scoring': { requires_consistency: true };
    'recommendation': { requires_reasoning: true };

    // Generation tasks
    'question_generation': { creativity: 'medium' };
    'memo_writing': { creativity: 'high' };
    'summary_writing': { creativity: 'low' };

    // Web research
    'web_search_synthesis': { requires_web: true };
  };

  // ─────────────────────────────────────────────────────────────────────────
  // ROUTING LOGIC
  // ─────────────────────────────────────────────────────────────────────────

  route: async (request: {
    task: TaskType;
    input: string;
    strategy?: RoutingStrategy;
    constraints?: {
      max_cost?: number;
      max_latency_ms?: number;
      min_quality?: number;
      preferred_models?: string[];
      excluded_models?: string[];
    };
  }) => {

    const strategy = request.strategy || 'balanced';

    // Filter models by constraints
    let candidates = this.models.registry
      .filter(m => m.status === 'active')
      .filter(m => !request.constraints?.excluded_models?.includes(m.id))
      .filter(m => {
        if (request.constraints?.preferred_models?.length) {
          return request.constraints.preferred_models.includes(m.id);
        }
        return true;
      });

    // Apply task-specific requirements
    const taskConfig = this.taskTypes[request.task];
    if (taskConfig.requires_vision) {
      candidates = candidates.filter(m => m.capabilities.supports_vision);
    }
    if (taskConfig.requires_long_context) {
      candidates = candidates.filter(m => m.capabilities.context_window > 100000);
    }

    // Exploration: occasionally try under-sampled models
    if (Math.random() < this.strategies.exploration.probability) {
      const exploration = this.strategies.exploration.select(request.task, candidates);
      if (exploration) {
        return { model: exploration, reason: 'exploration' };
      }
    }

    // Apply selected strategy
    const selected = this.strategies[strategy].select(request.task, candidates);

    if (!selected) {
      throw new Error(`No suitable model found for task ${request.task}`);
    }

    return { model: selected, reason: strategy };
  };

  // ─────────────────────────────────────────────────────────────────────────
  // FALLBACK HANDLING
  // ─────────────────────────────────────────────────────────────────────────

  fallback: {
    enabled: true;

    triggers: [
      'rate_limit_exceeded',
      'timeout',
      'server_error',
      'context_length_exceeded',
      'content_filter'
    ];

    strategy: async (failedModel: Model, task: TaskType, error: Error) => {
      // Mark model as potentially degraded
      await this.updateModelStatus(failedModel.id, 'degraded');

      // Find alternative
      const alternatives = this.models.registry
        .filter(m => m.id !== failedModel.id)
        .filter(m => m.status === 'active');

      // Route to next best
      return this.strategies.balanced.select(task, alternatives);
    };

    maxRetries: 3;
    backoffMs: [1000, 2000, 4000];
  };
}
```

```typescript
      // Unique insights (one model saw something others missed)
      unique_insights: responses.flatMap(r =>
        r.key_points.filter(p =>
          !otherResponsesContain(responses, r.model, p)
        )
      );

      // Overall assessment
      overall: {
        direction: determineOverallDirection(responses),
        confidence: calculateOverallConfidence(responses),
        needs_human_review: responses.length === 0 ||
          analysis.disagreements.significance === 'critical'
      };
    };
  };

  // ─────────────────────────────────────────────────────────────────────────
  // FUSION ENGINE
  // ─────────────────────────────────────────────────────────────────────────

  fuseResponses: async (analysis: ResponseAnalysis, context: EnrichedContext) => {

    // Use a capable model to synthesize
    const fusionModel = await this.modelRouter.route({
      task: 'synthesis',
      strategy: 'quality_first'
    });

    const fusionPrompt = `

${analysis.consensus.points.map(p => `- ${p}`).join('\n')}

${analysis.disagreements.points.map(p => `- ${p}`).join('\n')}

${analysis.unique_insights.map(i => `- [${i.model}] ${i.insight}`).join('\n')}

    const fusion = await callModel(fusionModel.model, fusionPrompt);
```

## 3.4 Internal Quality Engine (Background)

> **Note importante**: Cette section decrit des mecanismes 100% internes.
> L'utilisateur ne voit jamais cette complexite - il voit simplement des resultats de qualite.
> C'est la "tuyauterie" qui fait que le systeme fonctionne bien, pas un argument de vente.

```typescript
// ═══════════════════════════════════════════════════════════════════════════
// INTERNAL QUALITY ENGINE - Background Optimization (invisible to users)
// ═══════════════════════════════════════════════════════════════════════════

interface InternalQualityEngine {

  // ─────────────────────────────────────────────────────────────────────────
  // PERFORMANCE TRACKING
  // ─────────────────────────────────────────────────────────────────────────

  tracking: {
    // Every model call is logged
    logModelCall: (call: {
      model_id: string;
      task_type: TaskType;
      input_tokens: number;
      output_tokens: number;
      latency_ms: number;
      cost: number;
      success: boolean;
      error?: string;
    }) => void;

    // Quality feedback (from users or automated)
    logQualityFeedback: (feedback: {
      model_id: string;
      task_type: TaskType;
      call_id: string;

      // Ratings
      quality_score: number;        // 1-5 or 1-100
      accuracy_score?: number;      // If verifiable
      helpfulness_score?: number;

      // Source
      feedback_source: 'user_explicit' | 'user_implicit' | 'automated' | 'outcome';

      // Details
      comments?: string;
      corrections?: string;
    }) => void;

    // Outcome tracking (for investment decisions)
    logOutcome: (outcome: {
      deal_id: string;

      // What we predicted
      predictions: {
        recommendation: string;
        confidence: number;
        key_risks: string[];
        expected_multiple: number;
      };

      // What happened
      actual: {
        invested: boolean;
        current_status?: string;
        current_multiple?: number;
        learnings?: string[];
      };

      // Time
      prediction_date: Date;
      outcome_date: Date;
    }) => void;
  };

  // ─────────────────────────────────────────────────────────────────────────
  // PERFORMANCE AGGREGATION
  // ─────────────────────────────────────────────────────────────────────────

  aggregation: {
    // Run daily
    aggregatePerformance: async () => {
      for (const model of models) {
        for (const taskType of taskTypes) {
          const metrics = await calculateMetrics(model.id, taskType);

          await updateModelPerformance(model.id, taskType, {
            sample_size: metrics.sample_size,
            avg_quality_score: metrics.avg_quality,
            avg_latency_ms: metrics.avg_latency,
            success_rate: metrics.success_rate,
            cost_efficiency: metrics.avg_quality / metrics.avg_cost,
            trend: calculateTrend(metrics)
          });
        }
      }
    };

    // Detect performance changes
    detectAnomalies: async () => {
      const anomalies = [];

      for (const model of models) {
        const recent = await getRecentPerformance(model.id, '7d');
        const baseline = await getBaselinePerformance(model.id, '30d');

        // Quality drop?
        if (recent.avg_quality < baseline.avg_quality * 0.9) {
          anomalies.push({
            model: model.id,
            type: 'quality_drop',
            magnitude: (baseline.avg_quality - recent.avg_quality) / baseline.avg_quality,
            recommendation: 'reduce_traffic'
          });
        }

        // Latency spike?
        if (recent.avg_latency > baseline.avg_latency * 1.5) {
          anomalies.push({
            model: model.id,
            type: 'latency_spike',
            magnitude: (recent.avg_latency - baseline.avg_latency) / baseline.avg_latency,
            recommendation: 'monitor'
          });
        }
      }

      return anomalies;
    };
  };

  // ─────────────────────────────────────────────────────────────────────────
  // MODEL DIVERSITY LEARNING
  // ─────────────────────────────────────────────────────────────────────────

  diversityLearning: {
    // Track when models disagree
    trackDisagreement: (analysis: ResponseAnalysis) => {
      const disagreementPairs = [];

      for (let i = 0; i < analysis.individual_responses.length; i++) {
        for (let j = i + 1; j < analysis.individual_responses.length; j++) {
          const modelA = analysis.individual_responses[i].model;
          const modelB = analysis.individual_responses[j].model;
          const disagreed = hasDisagreement(
            analysis.individual_responses[i],
            analysis.individual_responses[j]
          );

          disagreementPairs.push({
            model_a: modelA,
            model_b: modelB,
            disagreed: disagreed,
            task_type: analysis.task_type
          });
        }
      }

      // Update diversity matrix
      for (const pair of disagreementPairs) {
        await updateDiversityMatrix(
          pair.model_a,
          pair.model_b,
          pair.disagreed,
          pair.task_type
        );
      }
    };

    // Use diversity matrix for ensemble selection
    selectDiverseModels: async (taskType: TaskType, count: number) => {
      const diversityMatrix = await getDiversityMatrix(taskType);

      // Greedy selection maximizing pairwise diversity
      const selected = [];
      const candidates = await getActiveModels();

      // Start with best performing model
      selected.push(
        candidates.sort((a, b) =>
          b.learned_performance.by_task[taskType]?.avg_quality_score -
          a.learned_performance.by_task[taskType]?.avg_quality_score
        )[0]
      );

      // Add models that maximize diversity from selected set
      while (selected.length < count && candidates.length > 0) {
        let maxDiversity = -1;
        let nextModel = null;

        for (const candidate of candidates) {
          if (selected.includes(candidate)) continue;

          const diversity = selected.reduce((sum, s) =>
            sum + (diversityMatrix[s.id]?.[candidate.id] || 0.5), 0
          ) / selected.length;

          if (diversity > maxDiversity) {
            maxDiversity = diversity;
            nextModel = candidate;
          }
        }

        if (nextModel) selected.push(nextModel);
      }

      return selected;
    };
  };

  // ─────────────────────────────────────────────────────────────────────────
  // PREDICTION CALIBRATION
  // ─────────────────────────────────────────────────────────────────────────

  calibration: {
    // "When we predict X confidence, how often are we right?"
    calculateCalibration: async () => {
      const outcomes = await getOutcomesWithPredictions();

      const buckets = {
        '90-100': { predicted: 0, correct: 0 },
        '70-89': { predicted: 0, correct: 0 },
        '50-69': { predicted: 0, correct: 0 },
        '30-49': { predicted: 0, correct: 0 },
        '0-29': { predicted: 0, correct: 0 }
      };

      for (const outcome of outcomes) {
        const bucket = getBucket(outcome.predicted_confidence);
        buckets[bucket].predicted++;
        if (outcome.was_correct) buckets[bucket].correct++;
      }

      return {
        calibration: Object.entries(buckets).map(([range, data]) => ({
          confidence_range: range,
          predicted_count: data.predicted,
          actual_accuracy: data.correct / data.predicted,
          expected_accuracy: getRangeMidpoint(range) / 100,
          calibration_error: Math.abs(
            data.correct / data.predicted - getRangeMidpoint(range) / 100
          )
        })),

        overall_calibration_error: calculateOverallCalibration(buckets),

        recommendation: generateCalibrationRecommendation(buckets)
      };
    };

    // Adjust confidence scores based on calibration
    adjustConfidence: (rawConfidence: number) => {
      const calibration = await getLatestCalibration();
      return applyCalibrationCurve(rawConfidence, calibration);
    };
  };
}
```
